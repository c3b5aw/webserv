257  bool _parse_configuration() {
258  std::string line;
259  std::stringstream ss(_conf_file);
260  Models::IBlock *current_block = NULL;
261 
262  int line_nbr = 0, scope = 0;
263  while (std::getline(ss, line) && ++line_nbr) {
264  int token = _resolve_line(&line);
265  switch (token) {
266  case CONF_BLOCK_ALLOWED_METHODS: {
267  _extract_value("allowed_methods", &line, false);
268  std::vector<std::string> split;
269  _split_string(line, ' ', &split);
270  current_block->deny_all_methods();
271  std::vector<std::string>::const_iterator it = split.begin();
272  for (; it != split.end(); it++) {
273  HTTP::METHODS m = HTTP::enumerate_method(*it);
274  if (m == HTTP::METH_UNKNOWN)
275  return unknown_method_error(*it);
276  if (m != HTTP::METH_GET && m != HTTP::METH_POST && m != HTTP::METH_DELETE)
277  return not_implemented_method_error(*it);
278  current_block->set_method(HTTP::enumerate_method(*it), true);
279  }
280  break;
281  }
282  case CONF_BLOCK_AUTOINDEX: {
283  _extract_value("autoindex", &line, false);
284  if (line != "on" && line != "off")
285  return invalid_value_error(line, line_nbr);
286  current_block->set_autoindex(line == "on");
287  break;
288  }
289  case CONF_BLOCK_BODY_LIMIT: {
290  _extract_value("body_limit", &line, false);
291  if (!_is_digits(line))
292  return invalid_value_error(line, line_nbr);
293  current_block->set_body_limit(atoi(line.c_str()));
294  break;
295  }
296  case CONF_BLOCK_CGI: {
297  _extract_value("cgi", &line, false);
298  if (line.find(" ") == std::string::npos)
299  return invalid_value_error(line, line_nbr);
300  std::string extension = line.substr(0, line.find(" "));
301  std::string cgi_path = line.substr(line.find(" ") + 1, line.size());
302  current_block->set_cgi(extension, cgi_path);
303  break;
304  }
305  case CONF_BLOCK_ERROR_PAGE: {
306  _extract_value("error_page", &line, false);
307 
308  if (line.size() < 6 || !_is_digits(line.substr(0, 3))
309  || line.find(" ") == std::string::npos
310  || line.substr(0, line.find(" ")).size() != 3)
311  return invalid_value_error(line, line_nbr);
312 
313  int error_code = atoi(line.substr(0, line.find(" ")).c_str());
314  std::string page_path = line.substr(line.find(" ") + 1, line.size());
315  std::string source;
316  if (!_dump_file(page_path, &source))
317  return invalid_value_error(page_path, line_nbr);
318  current_block->set_error_page(error_code, source);
319  break;
320  }
321  case CONF_BLOCK_INDEX: {
322  _extract_value("index", &line, false);
323 
324  std::vector<std::string> split;
325  _split_string(line, ' ', &split);
326 
327  if (split.size() == 0)
328  return invalid_value_error(line, line_nbr);
329  std::vector<std::string>::const_iterator it = split.begin();
330  for (; it != split.end(); it++) {
331  if (it->size() > 0)
332  current_block->add_index(*it);
333  }
334  break;
335  }
336  case CONF_SERVER_LOCATION: {
337  if (scope == 2)
338  return nested_locations_error(line, line_nbr);
339  ++scope;
340  _extract_value("location", &line, true);
341 
342  if (line.size() < 1 || line[0] != '/')
343  return invalid_value_error(line, line_nbr);
344  if (line[line.size() - 1] == ' ')
345  line = line.substr(0, line.size() - 1);
346  ILocation *block = _servers.back()->new_location(line);
347  if (!block)
348  return invalid_value_error(line, line_nbr);
349  current_block = block;
350  break;
351  }
352  case CONF_SERVER_LISTEN: {
353  _extract_value("listen", &line, false);
354  if (scope != 1)
355  return unexpected_token_line_error("listen", line_nbr);
356  if (line.size() == 0)
357  return invalid_value_error(line, line_nbr);
358 
359  if (line.find(":") == std::string::npos) {
360  if (!_is_digits(line)) {
361  if (line == "*")
362  _servers.back()->set_host("0.0.0.0");
363  else
364  _servers.back()->set_host(line);
365  } else {
366  _servers.back()->set_host("0.0.0.0");
367  _servers.back()->set_port(atoi(line.c_str()));
368  }
369  break;
370  } else {
371  if (line[0] == ':') {
372  _servers.back()->set_host("0.0.0.0");
373  _servers.back()->set_port(atoi(line.substr(1, line.size()).c_str()));
374  } else {
375  if (line.substr(0, line.find(":")) == "*")
376  _servers.back()->set_host("0.0.0.0");
377  else
378  _servers.back()->set_host(line.substr(0, line.find(":")));
379  _servers.back()->set_port(atoi(line.substr(
380  line.find(":") + 1, line.size()).c_str()));
381  }
382  break;
383  }
384  return invalid_value_error(line, line_nbr);
385  }
386  case CONF_BLOCK_REDIRECT: {
387  _extract_value("redirect", &line, false);
388 
389  if (line.find(" ") == std::string::npos)
390  return invalid_value_error(line, line_nbr);
391 
392  int status_code = atoi(line.c_str());
393  if (status_code < 200 || status_code > 527)
394  return invalid_http_code_error(status_code);
395 
396  std::string url = line.erase(0, 4);
397  current_block->set_redirection(url, status_code);
398  break;
399  }
400  case CONF_BLOCK_ROOT: {
401  _extract_value("root", &line, false);
402  if (line.size() == 0)
403  return invalid_value_error(line, line_nbr);
404  current_block->set_root(line);
405  break;
406  }
407  case CONF_SERVER_NAME: {
408  _extract_value("server_name", &line, false);
409  if (scope != 1)
410  return unexpected_token_line_error("server_name", line_nbr);
411  if (line.size() == 0)
412  return invalid_value_error(line, line_nbr);
413  _servers.back()->set_name(line);
414  break;
415  }
416  case CONF_BLOCK_UPLOAD_PASS: {
417  _extract_value("upload_pass", &line, false);
418 
419  if (line.size() == 0)
420  return invalid_value_error(line, line_nbr);
421  current_block->set_upload_pass(line);
422  break;
423  }
424  case CONF_EMPTY_TOKEN:
425  break;
426  case CONF_SERVER_OPENING: {
427  if (scope != 0)
428  return unexpected_token_line_error("server", line_nbr);
429  ++scope;
430  _servers.push_back(new IServer());
431  current_block = _servers.back();
432  break;
433  }
434  case CONF_ERRORENOUS_TOKEN:
435  return errorneous_line_error(line, line_nbr);
436  case CONF_BLOCK_CLOSING: {
437  if (scope <= 0)
438  return invalid_scope_error(line, line_nbr);
439  if (scope == 2)
440  current_block = _servers.back();
441  if (scope == 1)
442  current_block = 0;
443  --scope;
444  break;
445  }
446  default:
447  return errorneous_line_error(line, line_nbr);
448  }
449  }
450  if (scope != 0)
451  return invalid_value_error("EOF", line_nbr);
452  return true;
453  }
